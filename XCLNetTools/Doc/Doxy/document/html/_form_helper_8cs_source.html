243  public static string[] GetFromParamsValudByPre(string preName)
244  {
245  HttpContext context = HttpContext.Current;
246  IList<string> str = new List<string>();
247  if (!string.IsNullOrEmpty(preName))
248  {
249  NameValueCollection ps = context.Request.Params;
250  foreach (var m in ps.AllKeys)
251  {
252  if (m.IndexOf(preName) == 0)
253  {
254  str.Add(ps[m]);
255  }
256  }
257  }
258  return str.ToArray();
