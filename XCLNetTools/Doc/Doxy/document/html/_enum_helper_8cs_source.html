38  public static List<XCLNetTools.Enum.EnumFieldModel> GetEnumFieldModelList(Type emType)
39  {
40  if (!emType.IsEnum)
41  {
42  throw new InvalidOperationException();
43  }
44  List<XCLNetTools.Enum.EnumFieldModel> list = new List<XCLNetTools.Enum.EnumFieldModel>();
45  System.Reflection.FieldInfo[] fields = emType.GetFields();
46  foreach (FieldInfo field in fields)
47  {
48  if (field.FieldType.IsEnum)
49  {
50  XCLNetTools.Enum.EnumFieldModel model = new XCLNetTools.Enum.EnumFieldModel();
51  model.Value = ((int)emType.InvokeMember(field.Name, BindingFlags.GetField, null, null, null)).ToString();
52  model.Text = field.Name;
53  Object[] customObjs = field.GetCustomAttributes(typeof(DescriptionAttribute), false);
54  if (null != customObjs && customObjs.Length > 0)
55  {
56  model.Description = ((DescriptionAttribute)customObjs[0]).Description;
57  }
58  list.Add(model);
59  }
60  }
61  return list;
62  }

109  public static List<XCLNetTools.Entity.TextValue> GetList(Type type)

201  public static List<T> GetEnumListByBitValue<T>(int val)
202  {
203  var tp = typeof(T);
204  if (!tp.IsEnum || val < 0) return null;
205  List<T> lst = new List<T>();
206  var values = System.Enum.GetValues(typeof(T));
207  if (null != values && values.Length > 0)
208  {
209  T temp = default(T);
210  foreach (var m in values)
211  {
212  temp = (T)System.Enum.Parse(tp, Convert.ToString(m));
213  if ((val & Convert.ToInt32(temp)) == Convert.ToInt32(temp))
214  {
215  lst.Add(temp);
216  continue;
217  }
218  }
219  }
