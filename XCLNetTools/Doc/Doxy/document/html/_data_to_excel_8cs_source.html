61  public static void OutPutExcel(OutPutParamClass paramClass)
62  {
63  StringBuilder str = new StringBuilder();
64 
65  #region 合法性检测
66 
67  if (null == paramClass.ds || paramClass.ds.Tables.Count == 0)
68  {
69  str.Append("要导出的数据不能为空，导出失败！；");
70  }
71  if (null != paramClass.conTitle && null != paramClass.ds)
72  {
73  if (paramClass.conTitle.Length != paramClass.ds.Tables.Count)
74  {
75  str.Append("自定义的标题与要导出的数据源的数量不匹配，导出失败！；");
76  }
77  if (paramClass.conTitle.ToList().Distinct().Count() != paramClass.conTitle.Length)
78  {
79  str.Append("自定义的标题项不能重复，因为此项要作为Sheet的名称，导出失败！；");
80  }
81  }
82  if (null != paramClass.outPutClass && paramClass.outPutClass.Count > 0)
83  {
84  if (paramClass.tableName.Length != paramClass.ds.Tables.Count)
85  {
86  str.Append("表名与dataSet的table数量不一致，导出失败！；");
87  }
88  }
89  if (str.Length > 0)
90  {
91  throw new Exception(str.ToString());
92  }
93 
94  #endregion 合法性检测
95 
96  #region 是否指定被操作的工作薄
97 
98  Workbook workbook = null;
99  if (!string.IsNullOrEmpty(paramClass.WorkBookFilePath))
100  {
101  workbook = new Workbook(paramClass.WorkBookFilePath);
102  }
103  else
104  {
105  workbook = new Workbook();
106  }
107 
108  #endregion 是否指定被操作的工作薄
109 
110  for (int i = 0; i < paramClass.ds.Tables.Count; i++)
111  {
112  Worksheet sheet = workbook.Worksheets[i];
113 
114  if (null != paramClass.conTitle && paramClass.conTitle.Length > 0)
115  {
116  sheet.Name = paramClass.conTitle[i];
117  }
118 
119  if (i != paramClass.ds.Tables.Count - 1)
120  {
121  workbook.Worksheets.Add();
122  }
123 
124  DataTable dt = paramClass.ds.Tables[i];
125  List<string> dtColNameLst = new List<string>();
126  for (int k = 0; k < dt.Columns.Count; k++)
127  {
128  dtColNameLst.Add(dt.Columns[k].ColumnName);
129  }
130 
131  #region 写入列名
132 
133  List<string> newNamesLst = new List<string>();
134  if (null != paramClass.outPutClass && paramClass.outPutClass.Count > 0)
135  {
136  OutPutClass outPutModel = paramClass.outPutClass.First(k => k.tableName == paramClass.tableName[i].Trim());
137  foreach (var m in outPutModel.fields)
138  {
139  newNamesLst.Add(m.newName);
140  if (dtColNameLst.Contains(m.oldName))
141  {
142  dt.Columns[m.oldName].ColumnName = m.newName;
143  }
144  }
145  }
146  else
147  {
148  newNamesLst = dtColNameLst;
149  }
150 
151  #endregion 写入列名
152 
153  #region 向sheet中写入数据
154 
155  dt.AcceptChanges();
156  sheet.Cells.ImportDataTable(dt.DefaultView.ToTable("dtNew", true, newNamesLst.ToArray()), paramClass.IsShowFieldLine, paramClass.FirstRowIndex, paramClass.FirstColumnIndex, dt.Rows.Count + 1, dt.Columns.Count, true, "yyyy-MM-dd HH:mm:ss");
157  sheet.AutoFitColumns();
158 
159  #endregion 向sheet中写入数据
160 
161  #region 添加样式
162 
163  Cells cells = sheet.Cells;
164  if (paramClass.IsShowCustomLine)
165  {
166  cells[0, 0].Value = string.Format("数据导出：{0}；导出时间：{1}；记录总数：{2}", paramClass.conTitle[i], DateTime.Now, dt.Rows.Count);
167  Aspose.Cells.Style styleCell0 = cells[0, 0].GetStyle();
168  styleCell0.Font.Color = System.Drawing.Color.Red;
169  cells[0, 0].SetStyle(styleCell0);
170  }
171  if (paramClass.IsShowFieldLine)
172  {
173  Range range = sheet.Cells.CreateRange(1, 0, 1, newNamesLst.Count);
174  range.Name = "Range1";
175  Aspose.Cells.Style style = workbook.Styles[workbook.Styles.Add()];
176  style.HorizontalAlignment = TextAlignmentType.Center;
177  style.Font.Color = System.Drawing.Color.Blue;
178  style.Font.IsBold = true;
179  StyleFlag styleFlag = new StyleFlag();
180  styleFlag.All = true;
181  range.ApplyStyle(style, styleFlag);
182  }
183 
184  #endregion 添加样式
185  }
186  paramClass.GetWorkBook = workbook;
187 
188  #region 保存
189 
190  if (!string.IsNullOrEmpty(paramClass.CustomFileName))
191  {
192  workbook.Save(paramClass.CustomFileName, paramClass.SaveFormat);
193  }
194  if (paramClass.AutoDownLoad)
195  {
196  string fileName = string.Format("{0}_数据导出.xlsx", paramClass.fileTitle);
197  bool isFirefox = HttpContext.Current.Request.Browser.Type.ToLower().Contains("firefox");
198  fileName = isFirefox ? fileName : HttpUtility.UrlEncode(fileName, System.Text.Encoding.UTF8);
199  workbook.Save(HttpContext.Current.Response, fileName, ContentDisposition.Attachment, new OoxmlSaveOptions(SaveFormat.Xlsx));
200  }

202  #endregion 保存
