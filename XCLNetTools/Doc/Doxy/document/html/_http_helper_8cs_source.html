57  public HttpResult GetHtml(HttpItem item)
58  {
59  //返回参数
60  HttpResult result = new HttpResult();
61  try
62  {
63  //准备参数
64  SetRequest(item);
65  }
66  catch (Exception ex)
67  {
68  result = new HttpResult();
69  result.Cookie = string.Empty;
70  result.Header = null;
71  result.Html = ex.Message;
72  result.StatusDescription = "配置参数时出错：" + ex.Message;
73  return result;
74  }
75  try
76  {
77  #region 得到请求的response
78 
79  using (response = (HttpWebResponse)request.GetResponse())
80  {
81  result.StatusCode = response.StatusCode;
82  result.StatusDescription = response.StatusDescription;
83  result.Header = response.Headers;
84  if (response.Cookies != null) result.CookieCollection = response.Cookies;
85  if (response.Headers["set-cookie"] != null) result.Cookie = response.Headers["set-cookie"];
86  MemoryStream _stream = new MemoryStream();
87  //GZIIP处理
88  if (response.ContentEncoding != null && response.ContentEncoding.Equals("gzip", StringComparison.InvariantCultureIgnoreCase))
89  {
90  //开始读取流并设置编码方式
91  //new GZipStream(response.GetResponseStream(), CompressionMode.Decompress).CopyTo(_stream, 10240);
92  //.net4.0以下写法
93  _stream = GetMemoryStream(new GZipStream(response.GetResponseStream(), CompressionMode.Decompress));
94  }
95  else
96  {
97  //开始读取流并设置编码方式
98  //response.GetResponseStream().CopyTo(_stream, 10240);
99  //.net4.0以下写法
100  _stream = GetMemoryStream(response.GetResponseStream());
101  }
102  //获取Byte
103  byte[] ResponseByte = _stream.ToArray();
104  _stream.Close();
105  if (ResponseByte != null & ResponseByte.Length > 0)
106  {
107  //是否返回Byte类型数据
108  if (item.ResultType == ResultType.Byte) result.ResultByte = ResponseByte;
109  //从这里开始我们要无视编码了
110  if (encoding == null)
111  {
112  Match meta = Regex.Match(Encoding.Default.GetString(ResponseByte), "<meta([^<]*)charset=([^<]*)[\"']", RegexOptions.IgnoreCase);
113  string c = (meta.Groups.Count > 1) ? meta.Groups[2].Value.ToLower() : string.Empty;
114  if (c.Length > 2)
115  encoding = Encoding.GetEncoding(c.Trim().Replace("\"", "").Replace("'", "").Replace(";", "").Replace("iso-8859-1", "gbk"));
116  else
117  {
118  if (string.IsNullOrEmpty(response.CharacterSet)) encoding = Encoding.UTF8;
119  else encoding = Encoding.GetEncoding(response.CharacterSet);
120  }
121  }
122  //得到返回的HTML
123  result.Html = encoding.GetString(ResponseByte);
124  }
125  else
126  {
127  //得到返回的HTML
128  result.Html = "本次请求并未返回任何数据";
129  }
130  }
131 
132  #endregion 得到请求的response
133  }
134  catch (WebException ex)
135  {
136  //这里是在发生异常时返回的错误信息
137  response = (HttpWebResponse)ex.Response;
138  result.Html = ex.Message;
139  if (response != null)
140  {
141  result.StatusCode = response.StatusCode;
142  result.StatusDescription = response.StatusDescription;
143  }
144  }
145  catch (Exception ex)
146  {
147  result.Html = ex.Message;
148  }
149  if (item.IsToLower) result.Html = result.Html.ToLower();
150  return result;
151  }

157  private MemoryStream GetMemoryStream(Stream streamResponse)
158  {
159  MemoryStream _stream = new MemoryStream();
160  int Length = 256;
161  Byte[] buffer = new Byte[Length];
162  int bytesRead = streamResponse.Read(buffer, 0, Length);
163  while (bytesRead > 0)
164  {
165  _stream.Write(buffer, 0, bytesRead);
166  bytesRead = streamResponse.Read(buffer, 0, Length);
167  }
168  return _stream;

175  private void SetRequest(HttpItem item)
176  {
177  // 验证证书
178  SetCer(item);
179  //设置Header参数
180  if (item.Header != null && item.Header.Count > 0) foreach (string key in item.Header.AllKeys)
181  {
182  request.Headers.Add(key, item.Header[key]);
183  }
184  // 设置代理
185  SetProxy(item);
186  if (item.ProtocolVersion != null) request.ProtocolVersion = item.ProtocolVersion;
187  request.ServicePoint.Expect100Continue = item.Expect100Continue;
188  //请求方式Get或者Post
189  request.Method = item.Method;
190  request.Timeout = item.Timeout;
191  request.KeepAlive = item.KeepAlive;
192  request.ReadWriteTimeout = item.ReadWriteTimeout;
193  //Accept
194  request.Accept = item.Accept;
195  //ContentType返回类型
196  request.ContentType = item.ContentType;
197  //UserAgent客户端的访问类型，包括浏览器版本和操作系统信息
198  request.UserAgent = item.UserAgent;
199  // 编码
200  encoding = item.Encoding;
201  //设置Cookie
202  SetCookie(item);
203  //来源地址

205  //是否执行跳转功能

207  //设置Post数据

273  private void SetPostData(HttpItem item)
274  {
275  //验证在得到结果时是否有传入数据
276  if (request.Method.Trim().ToLower().Contains("post"))
277  {
278  if (item.PostEncoding != null)
279  {
280  postencoding = item.PostEncoding;
281  }
282  byte[] buffer = null;
283  //写入Byte类型
284  if (item.PostDataType == PostDataType.Byte && item.PostdataByte != null && item.PostdataByte.Length > 0)
285  {
286  //验证在得到结果时是否有传入数据
287  buffer = item.PostdataByte;
288  }//写入文件
289  else if (item.PostDataType == PostDataType.FilePath && !string.IsNullOrEmpty(item.Postdata))
290  {
291  StreamReader r = new StreamReader(item.Postdata, postencoding);
292  buffer = postencoding.GetBytes(r.ReadToEnd());
293  r.Close();
294  } //写入字符串
295  else if (!string.IsNullOrEmpty(item.Postdata))
296  {
297  buffer = postencoding.GetBytes(item.Postdata);
298  }
299  if (buffer != null)
300  {
301  request.ContentLength = buffer.Length;
302  request.GetRequestStream().Write(buffer, 0, buffer.Length);
303  }
304  }
305  }
